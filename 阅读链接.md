### webpack.config.js
```js
'use strict';
const webpack = require('webpack');
const HtmlWebpackPlugin = require('html-webpack-plugin');
const { CleanWebpackPlugin } = require('clean-webpack-plugin');
const CopyPlugin = require('copy-webpack-plugin');
const path = require('path');
const TsconfigPathsPlugin = require('tsconfig-paths-webpack-plugin');
const NodePolyfillPlugin = require("node-polyfill-webpack-plugin");

const CONTEXT_PATH = path.resolve(`${__dirname}/../..`);
const DIST_PATH = `${CONTEXT_PATH}/dist/webpack/`;
const SRC_PATH = `${CONTEXT_PATH}/src`;
const KB_LIMIT = 170000; // bundle.js size limit in kB

const { PACKAGE_CODE, LOCKE_USERPOOL_CODE } = process.env;
const tsConfigFile = `${CONTEXT_PATH}/config/tsconfig/tsconfig.${PACKAGE_CODE}.json`;

module.exports = {
  mode: 'production',
  context: CONTEXT_PATH,
  entry: [
    `${SRC_PATH}/polyfill.js`,
    `${SRC_PATH}/index.tsx`,
    `${SRC_PATH}/newrelic/index.ts`
  ],
  output: {
    publicPath: '/',
    path: DIST_PATH,
    filename: 'assets/js/bundle.[fullhash].min.js'
  },
  resolve: {
    extensions: ['.ts', '.tsx', '.js', '.json'],
    plugins: [new TsconfigPathsPlugin({ configFile: tsConfigFile })],
    alias: {
      buffer: 'buffer',
      crypto: 'crypto-browserify',
      stream: 'stream-browserify'
    }
  },
  module: {
    rules: [
      { test: /\.tsx?$/, loader: 'ts-loader', options: { configFile: tsConfigFile } },
      { enforce: 'pre', test: /\.js$/, loader: 'source-map-loader' }
    ]
  },
  performance: {
    hints: 'error',
    maxAssetSize: KB_LIMIT * 1024,
    maxEntrypointSize: KB_LIMIT * 1024,
  },
  plugins: [
    new NodePolyfillPlugin(),
    new CopyPlugin({
      patterns: [{
        from: `assets/.well-known/${PACKAGE_CODE}/`,
        to: `${DIST_PATH}/.well-known/`
      }, {
        from: `assets/.well-known/${PACKAGE_CODE}/apple-app-site-association`,
        to: `${DIST_PATH}/`
      }, {
        from: `assets/icons/`,
        to: `${DIST_PATH}/assets/icons/`
      }]
    }),
    new HtmlWebpackPlugin({
      template: `${SRC_PATH}/index.html`,
      inject: true,
      minify: {
        collapseWhitespace: true,
        removeComments: true,
        removeRedundantAttributes: true,
        removeScriptTypeAttributes: true,
        removeStyleLinkTypeAttributes: true,
        useShortDoctype: true,
        minifyJS: {
          ascii_only: true
        }
      }
    }),
    new CleanWebpackPlugin(),
    new webpack.DefinePlugin({
      'process.env.USERPOOL_CODE': JSON.stringify(LOCKE_USERPOOL_CODE || 'au')
    })
  ]
};
```
